class AnimatedBarChart:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.fig = plot.figure()
        plot.ylim(0, max(y) + 0.5*max(y))

        self.n = 1000  # Number of frames
        self.bar_collection = plot.bar(x, self.bar_heights_at_frame(1))

    def bar_heights_at_frame(self, i):
        heights = []
        for value in self.y:
            heights.append(value * (i / self.n))
        return heights

    def animate(self, i):
        y = self.bar_heights_at_frame(i + 1)
        for bar_number, bar in enumerate(self.bar_collection):
            bar.set_height(y[bar_number])

    def create(self):
        anim = FuncAnimation(self.fig, self.animate, repeat=False, blit=False, frames=self.n, interval=20)
        # plot.show()
        anim.save('movie.mp4', writer=FFMpegWriter(fps=60))
